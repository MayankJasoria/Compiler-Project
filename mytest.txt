declare module letsparty;

<<<driver program>>>
start
	** use swap once and the functions below once **
	declare m, n: integer;
	m := 1;
	n := 4;
	declare A: array[m..n] of integer;
	use module letsparty with parameters A, m;
	print(A);
end


<<module letsparty>>
takes input[A: array[1..4] of integer, B: integer];
start
	declare i:integer;
	for(i in 1..4)
	start
		A[i] := -(i*(i+1));
	end
	print(B);
end
