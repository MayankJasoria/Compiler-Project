
<<module compute_expr>>
takes input[ a: integer, b:integer, c:boolean];
returns [u:integer, v:boolean];
start
	u := a;
	v := c;
end

<<<driver program>>>
start
	declare x, y, z:integer;
	get_value(x);
	get_value(y);
	declare k, h:boolean;
	h:= false;
	[z, k]:= use module compute_expr with parameters x, y, h;				
	print(x);
	print(y);
	print(z);
	print(k);
end
